    <script>
        // Dados do pagamento (normalmente viriam do servidor)
        let paymentData = {
            payment_id: '',
            qr_code: '',
            qr_image_url: '',
            amount: 0,
            status: 'pending'
        };

        let timerSeconds = 1200; // 20 minutes
        let statusCheckInterval;

        // Inicializar página com dados do pagamento
        function initializePayment(data) {
            paymentData = data;
            document.getElementById('payment-id').textContent = data.payment_id;
            document.getElementById('amount').textContent = data.amount.toFixed(2).replace('.', ',');
            document.getElementById('qr-image').src = data.qr_image_url;

            // Iniciar timer
            startTimer();

            // Iniciar verificação de status
            startStatusCheck();
        }

        // Timer countdown
        function startTimer() {
            const timerInterval = setInterval(() => {
                timerSeconds--;
                
                const minutes = Math.floor(timerSeconds / 60);
                const seconds = timerSeconds % 60;
                
                document.getElementById('timer').textContent = 
                    `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;

                if (timerSeconds <= 0) {
                    clearInterval(timerInterval);
                    updateStatus('expired', '⏰ Pagamento expirado');
                }
            }, 1000);
        }

        // Verificar status do pagamento a cada 2 minutos
        function startStatusCheck() {
            // Primeira verificação após 30 segundos
            setTimeout(checkPaymentStatus, 30000);

            // Depois verificar a cada 2 minutos (120000ms = 120 segundos)
            statusCheckInterval = setInterval(checkPaymentStatus, 120000);
        }

        // Função para verificar status
        async function checkPaymentStatus() {
            try {
                const response = await fetch(`/php-pixgo/check-status.php?payment_id=${paymentData.payment_id}`);
                const data = await response.json();

                if (data.success) {
                    paymentData.status = data.status;

                    if (data.status === 'completed') {
                        clearInterval(statusCheckInterval);
                        updateStatus('completed', '✅ Pagamento confirmado! Obrigado pela compra.');
                        
                        // Redirecionar para página de sucesso após 3 segundos
                        setTimeout(() => {
                            window.location.href = '/success.html';
                        }, 3000);
                    } else if (data.status === 'expired') {
                        clearInterval(statusCheckInterval);
                        updateStatus('expired', '⏰ Pagamento expirado. Faça um novo pedido.');
                    } else if (data.status === 'cancelled') {
                        clearInterval(statusCheckInterval);
                        updateStatus('expired', '❌ Pagamento cancelado.');
                    }
                }
            } catch (error) {
                console.error('Erro ao verificar status:', error);
            }
        }

        // Atualizar status visual
        function updateStatus(status, message) {
            const statusContainer = document.getElementById('status-container');
            statusContainer.className = `status ${status}`;
            statusContainer.textContent = message;
        }

        // Copiar código Pix
        function copyPixCode() {
            navigator.clipboard.writeText(paymentData.qr_code);
            alert('Código Pix copiado com sucesso!');
        }
